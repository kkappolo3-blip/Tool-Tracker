        function restoreData(e) {
            const file = e.target.files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = (event) => {
                try { 
                    const data = JSON.parse(event.target.result);
                    if (data.tools) tools = data.tools;
                    if (data.plannings) plannings = data.plannings;
                    if (data.accounts) accounts = data.accounts;
                    saveToStorage(); 
                    populateAccountDropdowns();
                    showNotification('Data berhasil dipulihkan'); 
                }
                catch (err) { showNotification('File tidak valid'); }
            };
            reader.readAsText(file);
            e.target.value = '';
        }

        function setDropdownValue(selectId, wrapId, inputId, value) {
            const select = document.getElementById(selectId);
            const wrap = document.getElementById(wrapId);
            const input = document.getElementById(inputId);
            if(!select || !wrap || !input) return;
            
            let found = false;
            for (let i = 0; i < select.options.length; i++) { if (select.options[i].value === value) { found = true; break; } }
            if (found && value) { select.value = value; wrap.classList.add('hidden'); } 
            else if (value) { select.value = 'ADD_NEW'; wrap.classList.remove('hidden'); input.value = value; } 
            else { select.value = ''; wrap.classList.add('hidden'); }
        }

        function editTool(id) {
            const tool = tools.find(t => t.id === id);
            if (!tool) return;
            isEditing = true;
            switchTab('tools'); 
            document.getElementById('editId').value = tool.id;
            document.getElementById('itemName').value = tool.name;
            document.getElementById('itemDesc').value = tool.description || '';
            document.getElementById('itemUrl').value = tool.url || '';
            document.getElementById('toolStatus').value = tool.status;
            document.getElementById('toolVersion').value = tool.version || '';
            document.getElementById('toolReleaseDate').value = tool.releaseDate || '';
            
            handleStatusChange();
            setDropdownValue('toolCreatedWith', 'customCreatedWrap', 'customCreated', tool.createdWith);
            setAccountValue('toolCreatedAccount', 'customCreatedAccountWrap', 'customCreatedAccount', tool.createdByAccount);
            setDropdownValue('toolDeployWith', 'customDeployWrap', 'customDeploy', tool.deployWith);
            setAccountValue('toolDeployAccount', 'customDeployAccountWrap', 'customDeployAccount', tool.deployByAccount);
            
            document.getElementById('submitBtn').innerText = "Update Tool";
            document.getElementById('cancelEdit').classList.remove('hidden');
            
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        function deleteTool(id) {
            if (confirm('Hapus tool ini?')) { 
                tools = tools.filter(t => t.id !== id); 
                saveToStorage(); 
                showNotification('Tool dihapus'); 
            }
        }

        function editPlanning(id) {

            if (confirm('Hapus planning ini?')) {

                saveToStorage(); 
                showNotification('Planning dihapus'); 
            }
        }

        function convertToTool(id) {
            const plan = plannings.find(p => p.id === id);
            if (!plan) return;
            if (confirm('Pindahkan ke Tool Aktif?')) {
                const newTool = {
                    id: Date.now(),
                    name: plan.name,
                    description: plan.description,
                    url: plan.url,
                    status: 'Pending',
                    logs: plan.logs || [],
                    createdWith: '', createdByAccount: '', deployWith: '', deployByAccount: '', version: '', releaseDate: ''
                };
                tools.unshift(newTool);
                plannings = plannings.filter(p => p.id !== id);
                saveToStorage();
                showNotification('Dipindahkan ke Tool Aktif');
                switchTab('tools');
            }
        }

        function toggleLog(type, id) {
            const logContent = document.getElementById(`log-${type}-${id}`);
            const arrow = document.getElementById(`arrow-${type}-${id}`);
            if (!logContent) return;
            const isCollapsed = logContent.classList.contains('collapsed');
            logContent.classList.toggle('collapsed', !isCollapsed);
            logContent.classList.toggle('expanded', isCollapsed);
            if (arrow) arrow.innerText = isCollapsed ? '▲' : '▼';
        }

        // --- LOG MODAL & OPS ---
        function openLogModal(type, itemId, logId = null) {
            const modal = document.getElementById('logModal');
            const textarea = document.getElementById('logTextarea');
            const title = document.getElementById('modalTitle');
            
            const items = type === 'tool' ? tools : plannings;
            
            if (logId) {
                const item = items.find(i => i.id === itemId);
                const log = item?.logs?.find(l => l.id === logId);
                if (log) {
                    textarea.value = log.text;
                    title.innerText = 'Edit Catatan';
                    currentLogEdit = { type, itemId, logId };
                }
            } else {
                textarea.value = '';
                title.innerText = 'Tambah Catatan';
                currentLogEdit = { type, itemId };
            }
            
            modal.classList.add('active');
            setTimeout(() => textarea.focus(), 100);
        }

        function closeLogModal(event) {
            if (event && event.target !== event.currentTarget) return;
            const modal = document.getElementById('logModal');
            modal.classList.remove('active');
            currentLogEdit = null;
        }

        function saveLogFromModal() {
            if (!currentLogEdit) return;
            const text = document.getElementById('logTextarea').value.trim();
            if (!text) { showNotification('Catatan tidak boleh kosong'); return; }

            const { type, itemId, logId } = currentLogEdit;
            const items = type === 'tool' ? tools : plannings;
            const idx = items.findIndex(i => i.id === itemId);
            if (idx === -1) return;

            if (!items[idx].logs) items[idx].logs = [];

            if (logId) {
                const logIdx = items[idx].logs.findIndex(l => l.id === logId);
                if (logIdx !== -1) items[idx].logs[logIdx].text = text;
                showNotification('Catatan diperbarui');
            } else {
                items[idx].logs.unshift({ 
                    id: Date.now(), 
                    date: new Date().toLocaleDateString('id-ID', { day: 'numeric', month: 'short', year: 'numeric' }), 
                    text, completed: false
                });
                showNotification('Catatan ditambahkan');
            }

            saveToStorage();
            closeLogModal();
            setTimeout(() => {
                const logContent = document.getElementById(`log-${type}-${itemId}`);
                if (logContent) { logContent.classList.remove('collapsed'); logContent.classList.add('expanded'); }
            }, 100);
        }

        function toggleLogComplete(type, itemId, logId) {
            const items = type === 'tool' ? tools : plannings;
            const tIdx = items.findIndex(i => i.id === itemId);
            if (tIdx === -1) return;
            const lIdx = items[tIdx].logs.findIndex(l => l.id === logId);
            if (lIdx === -1) return;
            items[tIdx].logs[lIdx].completed = !items[tIdx].logs[lIdx].completed;
            saveToStorage();
        }

        function copyLogText(text) {
            navigator.clipboard.writeText(text).then(() => showNotification('Catatan disalin'))
            .catch(() => showNotification('Gagal menyalin'));
        }

        function deleteLog(type, itemId, logId) {
            if (confirm('Hapus catatan ini?')) {
                const items = type === 'tool' ? tools : plannings;
                const idx = items.findIndex(i => i.id === itemId);
                if (idx !== -1) { items[idx].logs = items[idx].logs.filter(l => l.id !== logId); saveToStorage(); showNotification('Catatan dihapus'); }
            }
        }

        function escapeHtml(text) {
            if (!text) return '';
            const div = document.createElement('div');
            div.textContent = text;
            return div.innerHTML;
        }
